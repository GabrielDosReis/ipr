set(TEST_BINARY unittests)

   main.cpp
   simple.cpp
)

   ${PROJECT_NAME}
   GTest::GTest
)

   PROPERTIES
      CXX_STANDARD 17
      CXX_STANDARD_REQUIRED ON
      CXX_EXTENSIONS OFF
)

   TARGET ${TEST_BINARY}
)
